Ubuntu 18.04

Login as root

apt upgrade
apt install apache2 -y
apt install mysql-server -y
apt install mysql-client -y
apt install libmysqlclient-dev -y
  
Change root password:
mysqladmin -u root -p OLDPASSWORD password NEWPASSWORD

Mysql cli:		mysql -u root -p
Create database:      	create database aruba;
Select database:	use aruba;
Create access user:	create user aruba;
Assign access rights:	grant all on aruba.* to 'aruba'@'localhost' identified by 'ArubaRocks';

Create table:
CREATE TABLE IF NOT EXISTS `devices` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `description` text,
  `ipaddress` text,
  `username` text,
  `password` text,
  `ostype` text,
  `platform` text,
  `osversion` text,
  `cpu` text,
  `memory` text,
  `sysinfo` longtext,
  `bridge` mediumtext,
  `interfaces` mediumtext,
  `vrf` mediumtext,
  `vsx` mediumtext,
  `vsf` mediumtext,
  `bps` mediumtext,
  `lldp` mediumtext,
  PRIMARY KEY (`id`)
) ENGINE=MyISAM AUTO_INCREMENT=31 DEFAULT CHARSET=latin1;
