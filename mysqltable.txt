CREATE TABLE IF NOT EXISTS `devices` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `description` text,
  `ipaddress` text,
  `username` text,
  `password` text,
  `ostype` text,
  `platform` text,
  `osversion` text,
  `cpu` text,
  `memory` text,
  `sysinfo` longtext,
  `bridge` mediumtext,
  `interfaces` mediumtext,
  `vrf` mediumtext,
  `vsx` mediumtext,
  `vsf` mediumtext,
  `bps` mediumtext,
  `lldp` mediumtext,
  PRIMARY KEY (`id`)
) ENGINE=MyISAM AUTO_INCREMENT=31 DEFAULT CHARSET=latin1;
